
ai_strategy_unify_india = {
    icon = "gfx/interface/icons/ai_strategy_icons/unification.dds"
    type = diplomatic

    possible = {
        NOT = {
            exists = c:BIC
        }
        is_country_capital_former_bic_territory = yes
        has_yet_to_unify_all_former_bic_territory = yes
    }

    wanted_army_ratio = {
        value = 150000
    }

    wanted_navy_ratio = {
        value = 400000
    }

    aggression = {
        if = {
            limit = {
                any_state_in_target_country_former_bic_territory = yes
                can_reach_target_country_by_land = yes
            }
            add = 1000
        }
    }

    strategic_region_scores = {
        region_north_india = {
            add = 1000
        }
        region_south_india = {
            add = 1000
        }
    }

    wargoal_scores = {
        annex_country = {
            if = {
                limit = {
                    NOT = {
                        is_subject_of = scope:target_state
                    }
                    any_state_in_target_country_former_bic_territory = yes
                    can_reach_target_state_by_land = yes
                }
                add = 1000
            }
        }
        return_state = {
            if = {
                limit = {
                    NOT = {
                        is_subject_of = scope:target_state
                    }
                    can_reach_target_state_by_land = yes
                }
                add = 1000
            }
        }
        conquer_state = {
            if = {
                limit = {
                    NOT = {
                        is_subject_of = scope:target_country
                    }
                    OR = {
                        any_state_in_target_country_former_bic_territory = yes
                    }
                    can_reach_target_state_by_land = yes
                }
                add = 1000
                if = {
                    limit = {
                        is_subject_of = scope:target_country
                    }
                    multiply = 0.0
                }
            }
        }
    }

    secret_goal_scores = {
        conquer = {
            if = {
                limit = {
                    NOT = {
                        is_subject_of = scope:target_state
                    }
                    OR = {
                        any_state_in_target_country_former_bic_territory = yes
                    }
                    can_reach_target_country_by_land = yes
                }
                add = 1000
            }
            if = {
                limit = {
                    is_subject_of = scope:target_country
                }
                multiply = 0.0
            }
        }
    }

    weight = {
        value = 1000
    }
}
