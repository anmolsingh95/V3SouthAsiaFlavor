can_reach_target_country_by_land = {
	has_diplomatic_relevance = scope:target_country
	any_scope_state = {
		any_neighbouring_state = {
			owner = this.owner
		}
	}
	has_strategic_land_adjacency = scope:target_country
}

can_reach_target_state_by_land = {
	has_diplomatic_relevance = scope:target_state.owner
	any_neighbouring_state = {
		owner = this.owner
	}
}

is_country_former_bic_subject = {
	has_global_variable_list = former_bic_subjects_list
	is_target_in_global_variable_list = {
		name = former_bic_subjects_list
		target = ROOT
	}
}
