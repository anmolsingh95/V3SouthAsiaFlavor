can_reach_target_country_by_land = {
	has_diplomatic_relevance = scope:target_country
	any_scope_state = {
		any_neighbouring_state = {
			owner = this.owner
		}
	}
	has_strategic_land_adjacency = scope:target_country
}

can_reach_target_state_by_land = {
	has_diplomatic_relevance = scope:target_state.owner
	any_neighbouring_state = {
		owner = this.owner
	}
}

is_country_capital_former_bic_territory = {
	has_global_variable_list = former_bic_territories_list
	is_target_in_global_variable_list = {
		list = former_bic_territories_list
		target = ROOT.capital
	}
}

has_yet_to_unify_all_former_bic_territory = {
	has_global_variable_list = former_bic_territories_list
	any_in_global_list = {
		name = former_bic_territories_list
		NOT = {
			owner = ROOT
		}
	}
}

is_target_country_capital_former_bic_territory = {
	has_global_variable_list = former_bic_territories_list
	is_target_in_global_variable_list = {
		name = former_bic_territories_list
		target = scope:target_country.capital
	}
}

is_target_state_former_bic_territory = {
	has_global_variable_list = former_bic_territories_list
	is_target_in_global_variable_list = {
		name = former_bic_territories_list
		target = scope:target_state
	}
}
